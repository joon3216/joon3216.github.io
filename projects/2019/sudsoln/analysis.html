count_milli <- function(x) {
    hours <- as.numeric(substr(x, 1, 1))
    minutes <- as.numeric(substr(x, 3, 4))
    seconds <- as.numeric(substring(x, 6))
    hours * 3600 + minutes * 60 + seconds
}
